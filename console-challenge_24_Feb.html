1. Basic Arithmetic Operations

Fundamentals

Performs addition, subtraction, multiplication, and division on two numbers. 

let a = 24;
let b = 6;

console.log("Addition:",       a + b);
console.log("Subtraction:",    a - b);
console.log("Multiplication:", a * b);
console.log("Division:",       a / b);
console.log("Modulus:",        a % b);

Expected Output

Addition: 30
Subtraction: 18
Multiplication: 144
Division: 4
Modulus: 0

2. Calculator with Functions

Functions

Reusable functions for each arithmetic operation.

function add(x, y)      { return x + y; }
function subtract(x, y) { return x - y; }
function multiply(x, y) { return x * y; }
function divide(x, y) {
  if (y === 0) return "Error: Division by zero";
  return x / y;
}

console.log("add(10, 5) =",      add(10, 5));
console.log("subtract(10, 5) =", subtract(10, 5));
console.log("multiply(10, 5) =", multiply(10, 5));
console.log("divide(10, 5) =",   divide(10, 5));
console.log("divide(10, 0) =",   divide(10, 0));

Expected Output

add(10, 5) = 15
subtract(10, 5) = 5
multiply(10, 5) = 50
divide(10, 5) = 2
divide(10, 0) = Error: Division by zero

3. Calculator using Switch-Case

Control Flow

Uses a switch statement to select the operation based on an operator string. 

function calculate(num1, operator, num2) {
  switch (operator) {
    case "+": return num1 + num2;
    case "-": return num1 - num2;
    case "*": return num1 * num2;
    case "/":
      if (num2 === 0) return "Error: Division by zero";
      return num1 / num2;
    default: return "Error: Invalid operator";
  }
}

console.log("15 + 3 =", calculate(15, "+", 3));
console.log("15 - 3 =", calculate(15, "-", 3));
console.log("15 * 3 =", calculate(15, "*", 3));
console.log("15 / 3 =", calculate(15, "/", 3));
console.log("15 ^ 3 =", calculate(15, "^", 3));

Expected Output

15 + 3 = 18
15 - 3 = 12
15 * 3 = 45
15 / 3 = 5
15 ^ 3 = Error: Invalid operator

4. Interactive Calculator with prompt()

User Input

Takes user input via browser prompts and computes the result.

let num1     = parseFloat(prompt("Enter first number:"));
let operator = prompt("Enter operator (+, -, *, /):");
let num2     = parseFloat(prompt("Enter second number:"));

let result;
switch (operator) {
  case "+": result = num1 + num2; break;
  case "-": result = num1 - num2; break;
  case "*": result = num1 * num2; break;
  case "/":
    result = num2 !== 0 ? num1 / num2 : "Error: Division by zero";
    break;
  default: result = "Invalid operator";
}

console.log(`${num1} ${operator} ${num2} = ${result}`);

Sample Output (for inputs: 8, *, 7)

8 * 7 = 56

5. Advanced: Power, Square Root & Percentage

Math Object

Uses the built-in Math object for advanced calculator operations. 

function power(base, exp) {
  return Math.pow(base, exp);
}

function squareRoot(n) {
  if (n < 0) return "Error: Negative input";
  return Math.sqrt(n);
}

function percentage(value, total) {
  return (value / total) * 100;
}

console.log("2 ^ 10 =",          power(2, 10));
console.log("sqrt(144) =",       squareRoot(144));
console.log("sqrt(-4) =",        squareRoot(-4));
console.log("45 out of 60 =",    percentage(45, 60) + "%");

Expected Output

2 ^ 10 = 1024
sqrt(144) = 12
sqrt(-4) = Error: Negative input
45 out of 60 = 75%
